016package net.milkbowl.vault2.economy;
017
018import org.jetbrains.annotations.NotNull;
019
020import java.math.BigDecimal;
021
022/**
023 * Indicates a typical Return for an Economy method. It includes a
024 * {@link ResponseType} indicating whether the plugin currently being used for
025 * Economy actually allows the method, or if the operation was a success or
026 * failure.
027 *
028 */
029public class EconomyResponse {
030
031    /**
032     * Enum for types of Responses indicating the status of a method call.
033     */
034    public enum ResponseType {
035        SUCCESS(1),
036        FAILURE(2),
037        NOT_IMPLEMENTED(3);
038
039        private int id;
040
041        ResponseType(final int id) {
042            this.id = id;
043        }
044
045        int getId() {
046            return id;
047        }
048    }
049
050    /**
051     * Amount modified by calling method
052     */
053    public final BigDecimal amount;
054    /**
055     * New balance of account
056     */
057    public final BigDecimal balance;
058    /**
059     * Success or failure of call. Using Enum of ResponseType to determine valid
060     * outcomes
061     */
062    public final ResponseType type;
063    /**
064     * Error message if the variable 'type' is ResponseType.FAILURE
065     */
066    public final String errorMessage;
067
068    /**
069     * Constructor for EconomyResponse
070     * @param amount Amount modified during operation
071     * @param balance New balance of account
072     * @param type Success or failure type of the operation
073     * @param errorMessage Error message if necessary (commonly null)
074     */
075    public EconomyResponse(@NotNull final BigDecimal amount, @NotNull final BigDecimal balance, @NotNull final ResponseType type, @NotNull final String errorMessage) {
076        this.amount = amount;
077        this.balance = balance;
078        this.type = type;
079        this.errorMessage = errorMessage;
080    }
081
082    /**
083     * Checks if an operation was successful
084     * @return Value
085     */
086    public boolean transactionSuccess() {
087        switch (type) {
088        case SUCCESS:
089            return true;
090        default:
091            return false;
092        }
093    }
094}
