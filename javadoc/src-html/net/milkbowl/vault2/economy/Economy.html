468    /**
469     *
470     * Sets the amount of monies for a player.
471     *
472     * @param pluginName the name of the plugin setting the currency
473     * @param accountID the unique identifier of the player's account
474     * @param amount the amount of currency to set for the player in the specified world
475     * @return an EconomyResponse object indicating the result of the operation
476     */
477    default EconomyResponse set(@NotNull final String pluginName, @NotNull final UUID accountID, @NotNull final BigDecimal amount) {
478
479        final BigDecimal balance = balance(pluginName, accountID);
480        final int compare = balance.compareTo(amount);
481        if(compare > 0) {
482            return withdraw(pluginName, accountID, balance.subtract(amount));
483        }
484
485        if(compare < 0) {
486            return deposit(pluginName, accountID, amount.subtract(balance));
487        }
488
489        return new EconomyResponse(BigDecimal.ZERO, amount, ResponseType.SUCCESS, "");
490    }
